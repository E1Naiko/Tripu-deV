  ORG 1000H
    MSJ DB "ESCRIBIR UN VECTOR DE 20 NUMEROS"
    MSJ2 DB "Ingresar un dígito del 0 al 9"
    MSJ3 DB "Carácter inválido: solo dígitos del 0 al 9"
    VECTOR DB ?
    NUMEROS DB "0123456789"
    LINEA DB 10
    FIN DB ?
    MSJ4 DB "VECTOR CARGADO:"
    ASD DB ?
  ORG 1500H
    CHAR DB ?
    
  ORG 3000H
SALTO:
    MOV BX, OFFSET LINEA
    MOV AL,1
    INT 7
RET
  
ES_DIGITO:
    PUSH BX
    PUSH AX
    MOV DH,0
    MOV BX, OFFSET NUMEROS
    MOV AL, 10
  RECORRER:
    CMP AH, BYTE PTR [BX]
    JZ ES_DIG
    INC BX
    DEC AL
    CMP AL,0
    JZ FIN_ES
    JMP RECORRER
  ES_DIG:
    MOV DH,1
  FIN_ES:
    POP AX
    POP BX
RET

LEER_DIGITO:
    PUSH BX
    PUSH AX
    PUSH DX
    CALL SALTO          ; SALTO DE LINEA
    MOV BX, OFFSET MSJ2
    MOV AL, OFFSET MSJ3-OFFSET MSJ2
    INT 7 
    MOV BX, OFFSET CHAR    ; CARACTER A LEER
    MOV AL,1
    INT 6
    MOV AH, BYTE PTR [BX]       ;PONGO EL CARACTER LEIDO EN AH
    CALL ES_DIGITO
    CMP DH,0                ; SI ES 0 NO ES NUMERO
    JZ NO_DIG               ; IMPRIMO MENSAJE DE QUE NO ES DIGITO Y SALGO DE LA SUBRUTINA
    MOV CL, AH              ; EN CH PONGO EL VALOR LEIDO
    JMP FIN_DIG             ; TERMINO LA SUBRUTINA
  NO_DIG:
    CALL SALTO
    MOV BX, OFFSET MSJ3
    MOV AL, OFFSET VECTOR-OFFSET MSJ3
    INT 7
  FIN_DIG:
    POP DX
    POP AX
    POP BX
RET

LEER_VECTOR:
  PUSH AX
  PUSH BX
  PUSH CX
  PUSH DX
  MOV DH,0       ;CONTADOR PARA SABER SI TERMINO LOS 10 NUM
LOOPE:
  MOV CL, BYTE PTR [BX]     ; en CL PONGO EL VALOR LOS ELEMENTOS DEL VECTOR
  MOV AL, 1
  CALL LEER_DIGITO
  MOV BYTE PTR[BX], CL      ; Reemplazo el valor en el vector
  CMP DH, AH
  JZ TERMINO
  INC BX                    ; AVANZO
  INC DH
  JMP LOOPE
TERMINO:
  POP DX
  POP CX
  POP BX
  POP AX
RET

  ORG 2000H
  MOV BX, OFFSET MSJ
  MOV AL, OFFSET MSJ2- OFFSET MSJ
  INT 7
  MOV BX, OFFSET VECTOR
  MOV AH, 3
  CALL LEER_VECTOR
  CALL SALTO
  MOV BX, OFFSET MSJ4
  MOV AL, OFFSET ASD- OFFSET MSJ4
  INT 7
  CALL SALTO
  MOV BX, OFFSET VECTOR
  MOV AL, OFFSET NUMEROS-OFFSET VECTOR
  INT 7
  HLT  
  END
