Const FIN= 'xxx'; MarcaBuscada='ford';
Modelo='fiesta'; colorBuscado=5;

  dias=1..31; meses=1..12;
  codcolor= 1..10;

  fecha= record

            mes:meses;
         end;

  auto= record
           marca:cadena;
           modelo:cadena;
           color:codcolor;
           fing:fecha;
         end;

var a: auto;
Begin
  leer(a);
  while (a.marca<>FIN) do

     if (a.marca=MarcaBuscada) then
                        Begin

                           then   cantTot:= cantTot+1;

     leer(a);
  end;

End;
