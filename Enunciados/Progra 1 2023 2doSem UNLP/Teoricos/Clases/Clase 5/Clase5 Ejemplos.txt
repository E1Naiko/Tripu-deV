Program Clase5eje2;
uses crt;
Type
  letras = set of char;
var
  vocales,cons:letras; letra:char;
  cantV,cantC:integer;
  r:real;
Begin
  clrscr;
  r:=54.45;
  writeln(r);
  cantV:=0;
  cantC:=0;
  vocales:=['a','e','i','o','u'];
  cons:= ['a'..'z'] - vocales;
  writeln('Ingrese un caracter');
  readln(letra);
  while (letra <> '@') do
  begin
     if (letra in vocales)
     then
         cantV:= cantV+1
     else
        if(letra in cons) then cantC:= cantC+1;
     writeln('Ingrese un caracter');
     readln(letra);
   end;
   Writeln ('La cantidad de vocales encontradas es: ',cantV);
   Writeln ('La cantidad de consonantes encontradas es: ',cantC);
   readkey;
End.

end;
