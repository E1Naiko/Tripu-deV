org 1000h
vocales db 'a','e','i','o','u'
cadena db "holamundo"
res db 0,0,0,0,0

org 3000h
CONTARCARACTER: PUSH BX
PUSH AX
PUSH DX
MOV BX,SP
ADD BX,8
MOV BX,[BX]
MOV DH,0 ;INICIALIZO CONTADOR

BUCLE: MOV AL,[BX] ;TRAIGO EL CARACTER A AL
CMP AL, "%"
JZ GUARDAR
CMO AL, AH ;COMPARO CON LA VOCAL
JNZ NO-CONTAR
INC DH ;SI ES IGUAL CUENTO

NO-CONTAR: INC BX
JMP BUCLE

GUARDAR: MOV AL,DH ;GUARDO EL CONTADOR EN AX
POP DX,PARA TRAER LA DIR DE GUARDADO 
MOV BX,DX
MOV [BX], AL ;GUARDO EL RESULTADO
POP AX
POP BX
RET

ORG 2000H
MOV CX, OFFSET CADENA
MOV BX, OFFSET VOCALES
MOV DH, OFFSET RESULTADO

ITERO: CMP BX,CX ;LLEGE AL FINAL DE VOCALES
JZ FIN
MOV AH,[BX]  ;TRAIGO EL VOCAL
PUSH CX
CALL CONCAR
POP CX
INC BX
INC DH
JMP ITERO
FIN: HTL
END
